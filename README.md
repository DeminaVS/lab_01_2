# 34 Вводится список из n годовых вкладов, предлагаемых банками, в формате: Банк Сумма Процент где: все значения разделены пробелом и сами не содержат пробелов; наименование банка уникально; Сумма : сумма для открытия вклада в руб. (целое число, >0); Процент : годовой процент по вкладу (вещественное число, (0,100]). Сохраните введенные данные в виде списка словарей: [ {"name": "Банк 1", "initial_sum": 50000, "rate": 5.2}, ... ] Далее определите (гарантируется, что искомый банк - один): самый доступный банк (с наименьшей первоначальной суммой); самый выгодный банк, принимая, что за год прибыль = сумма * процент / 100 . При выводе финансовых значений оставьте два знака после запятой.
```
n = int(input())

# Создаем список для хранения данных о вкладах
deposits = []

# Вводим данные о вкладах
for i in range(n):
    data = input().split()
    bank_name = data[0]
    initial_sum = int(data[1])
    rate = float(data[2])

    deposit = {
        "name": bank_name,
        "initial_sum": initial_sum,
        "rate": rate
    }
    deposits.append(deposit)

# Находим самый доступный банк (с наименьшей первоначальной суммой)
min_bank = deposits[0]
for bank in deposits:
    if bank["initial_sum"] < min_bank["initial_sum"]:
        min_bank = bank

# Находим самый выгодный банк (с максимальной прибылью)
max_profit_bank = deposits[0]
max_profit = max_profit_bank["initial_sum"] * max_profit_bank["rate"] / 100

for bank in deposits:
    profit = bank["initial_sum"] * bank["rate"] / 100
    if profit > max_profit:
        max_profit = profit
        max_profit_bank = bank

# Выводим результаты
print(f'Самый доступный банк: {min_bank["name"]}')
print(f'Самый выгодный банк: {max_profit_bank["name"]}')
print(f'Доходность: {max_profit:.2f}')
```
